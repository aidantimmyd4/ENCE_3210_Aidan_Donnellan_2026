// Q1. Write a C program that to print numbers from 0 to 10 and 10 to 0 using two while loops
#include <stdio.h>
// Prompted and assisted by Gemini

int count = 0;

int main() 
{
    // use two while loops    
    while (count <= 10){
        printf("%d\n", count);
        count++; // counts up from initialized count
    }
    
    count = 10; // reset count to 10
    printf("\n");
    
    while(count >= 0){
        printf("%d\n", count);
        count--; // counts down from initialized count
    }
    
    return 0;
}
